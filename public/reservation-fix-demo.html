Código de la Solución
✅ Validación Añadida al Manejador del Botón:
// Validar selectedTable antes de proceder
if (!selectedTable) {
    App.showAlert('danger', 'Error: No se ha seleccionado ninguna mesa...');
    confirmBtn.disabled = false;
    confirmBtn.innerHTML = originalText;
    return;
}

if (!selectedTable.id) {
    App.showAlert('danger', 'Error: Información de mesa inválida...');
    confirmBtn.disabled = false;
    confirmBtn.innerHTML = originalText;
    return;
}

try {
    // Envío del formulario con manejo de errores
    const formData = new FormData(form);
    formData.append('selected_table_id', selectedTable.id);
    formData.append('ajax', '1');
    
    App.submitFormAjaxWithData(form, formData);
} catch (error) {
    console.error('Error submitting reservation:', error);
    App.showAlert('danger', 'Error al procesar la reservación...');
    
    // Restaurar estado del botón
    confirmBtn.disabled = false;
    confirmBtn.innerHTML = originalText;
}
